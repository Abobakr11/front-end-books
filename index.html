الدرس التاني 

// تعريف نوع المستند بأنه HTML5
<!DOCTYPE html>
<html>
  <head>
    // تعيين الترميز لدعم جميع اللغات
    <meta charset="UTF-8" />
    // عنوان الصفحة
    <title>هذا كتاب html</title>
    // وصف للصفحة لمحركات البحث
    <meta name="description" content="هذا كتاب html" />
    // مكان إضافة أكواد CSS
    <style></style>
    // مكان إضافة أكواد JavaScript
    <script></script>
    // ربط صفحة CSS خارجية
    <link rel="stylesheet" href="">
  </head>
  <body>
    // نص يظهر في الصفحة
    هذا كتاب html
  </body>
</html>

الدرس التالت

// تعريف نوع المستند
<!DOCTYPE html>
<html>
  <head>
    // الترميز لدعم جميع اللغات
    <meta charset="UTF-8" />
    // عنوان الصفحة
    <title>هذا كتاب html</title>
    // وصف الصفحة
    <meta name="description" content="هذا كتاب html" />
  </head>
  <body>
    // عنوان رئيسي
    <h1>هذا كتاب html</h1>
    // عنوان فرعي كبير
    <h2>هذا كتاب html</h2>
    // عنوان فرعي متوسط
    <h3>هذا كتاب html</h3>
    // عنوان فرعي أصغر
    <h4>هذا كتاب html</h4>
    // عنوان فرعي صغير جدًا
    <h5>هذا كتاب html</h5>
    // أصغر عنوان فرعي
    <h6>هذا كتاب html</h6>
    // نص عادي
    هذا كتاب html
  </body>
</html>
